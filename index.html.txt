        // Meldung anzeigen
        function showMessage(text, showSpinner = false, duration = 3000, isError = false) {
            const message = document.getElementById('message');
            const messageText = document.getElementById('messageText');
            const spinner = document.getElementById('messageSpinner');
            
            messageText.textContent = text;
            
            if (showSpinner) {
                spinner.style.display = 'inline-block';
            } else {
                spinner.style.display = 'none';
            }
            
            if (isError) {
                message.classList.add('error');
            } else {
                message.classList.remove('error');
            }
            
            message.classList.add('show');
            
            if (duration > 0) {
                setTimeout(() => {
                    message.classList.remove('show');
                }, duration);
            }
            
            return {
                hide: () => {
                    message.classList.remove('show');
                },
                update: (newText) => {
                    messageText.textContent = newText;
                }
            };
        }

        // Daten als Textdatei herunterladen
        function downloadTextFile(text, filename) {
            // Blob erstellen
            const blob = new Blob([text], { type: 'text/plain;charset=utf-8' });
            
            // Download-Link erstellen
            const link = document.createElement('a');
            link.href = URL.createObjectURL(blob);
            link.download = filename;
            
            // Link zum DOM hinzufügen und klicken
            document.body.appendChild(link);
            link.click();
            
            // Link wieder entfernen
            document.body.removeChild(link);
        }
        
        // Daten generieren und anzeigen
        function generateTextData() {
            try {
                // Meldung anzeigen
                const loadingMsg = showMessage('Daten werden vorbereitet...', true, 0);
                
                // Daten formatieren
                const textData = formatDataAsText();
                
                // Dateiname generieren
                const fileName = generateFileName();
                
                // Text in Vorschau anzeigen
                document.getElementById('textPreview').textContent = textData;
                
                // Modal anzeigen
                showModal('textModal');
                
                // Download-Button konfigurieren
                document.getElementById('downloadBtn').onclick = function() {
                    downloadTextFile(textData, fileName);
                    showMessage('Datei wurde heruntergeladen!', false, 3000);
                };
                
                // Meldung aktualisieren
                loadingMsg.update('Daten erfolgreich vorbereitet!');
                setTimeout(() => {
                    loadingMsg.hide();
                }, 2000);
                
            } catch (error) {
                console.error('Fehler bei der Datengenerierung:', error);
                showMessage('Fehler bei der Datengenerierung: ' + error.message, false, 5000, true);
            }
        }
